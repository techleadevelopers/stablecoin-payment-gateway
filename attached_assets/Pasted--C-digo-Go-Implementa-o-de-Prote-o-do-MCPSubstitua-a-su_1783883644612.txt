import (
	"crypto/sha256"
	"encoding/hex"
	"errors"
	"net/http"

type OrderResponse struct {
	ID                uint            `json:"id"`
	Amount            decimal.Decimal `json:"amount"`
	OverpaymentAmount decimal.Decimal `json:"overpayment_amount"`
	Asset             string          `json:"asset"`
	Status            string          `json:"status"`
}

// HashApiKey gera um hash SHA-256 seguro para buscar no banco sem expor a chave em texto plano
func HashApiKey(apiKey string) string {
	hash := sha256.Sum256([]byte(apiKey))
	return hex.EncodeToString(hash[:])
}

// ToolGetOrderStatus agora exige a carteira do agente autenticado para validar ownership (Anti-IDOR)
func (s *MCPServer) ToolGetOrderStatus(orderID uint, authenticatedAgentWallet string) (*OrderResponse, error) {
	var order OrderResponse
	
	// Corrigido: A query obriga o match do ID E da carteira do agente dono da ordem
	err := s.db.QueryRow(
		`SELECT id, amount, overpayment_amount, asset, status 
		 FROM orders 
		 WHERE id = $1 AND agent_wallet = $2`, 
		orderID, authenticatedAgentWallet,
	).Scan(&order.ID, &order.Amount, &order.OverpaymentAmount, &order.Asset, &order.Status)

	if err != nil {
		return nil, errors.New("order not found or unauthorized access")
	}

	return &order, nil
}

// ListWebhookSubscriptions filtrado estritamente por agente
func (s *MCPServer) ListWebhookSubscriptions(apiKey string) ([]string, error) {
	apiKeyHash := HashApiKey(apiKey)
	
	rows, err := s.db.Query(
		`SELECT target_url FROM mcp_webhook_subscriptions WHERE agent_api_key_hash = $1`, 
		apiKeyHash,
	)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var urls []string
	for rows.Next() {
		var url string
		if err := rows.Scan(&url); err == nil {
			urls = append(urls, url)
		}
	}
	return urls, nil
}

func GetRequiredConfirmations(chainName string) uint64 {
	// Pega o valor configurado na ENV
	envVal := os.Getenv("ONCHAIN_CONFIRMATIONS_" + chainName)
	configVal, err := strconv.ParseUint(envVal, 10, 64)
	
	// Define os pisos mínimos de segurança incontornáveis para produção real
	var floor uint64
	switch chainName {
	case "POLYGON":
		floor = 64 // Proteção robusta contra reorgs agressivos da Polygon
	case "BSC":
		floor = 3  // Mínimo padrão seguro para BSC fast blocks
	default:
		floor = 12 // Fallback defensivo para outras chains EVM
	}

	if err != nil || configVal < floor {
		return floor // Override forçado para o piso seguro
	}
	
	return configVal
}
