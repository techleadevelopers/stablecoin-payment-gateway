package adversarial

import (
	"bytes"
	"context"
	"crypto/sha256"
	"encoding/hex"
	"encoding/json"
	"fmt"
	"net/http"
	"net/http/httptest"
	"sync"
	"testing"
	"time"
)

type HybridAdversarialSuite struct {
	baseURL string
	client  *http.Client
}

func NewHybridAdversarialSuite(baseURL string) *HybridAdversarialSuite {
	return &HybridAdversarialSuite{
		baseURL: baseURL,
		client:  &http.Client{Timeout: 3 * time.Second},
	}
}

// ---------------------------------------------------------------------
// 1. MODULO HUMAN RAIL: Ataque de Envenenamento e Concorrência de Webhook
// ---------------------------------------------------------------------
// Tenta forçar múltiplos payouts Pix enviando notificações idênticas ou em 
// descompasso de milissegundos para simular falhas no provedor fiduciário (Efí).
func (s *HybridAdversarialSuite) AttackHumanRailWebhook(ctx context.Context, txHash string) (int, int) {
	url := fmt.Sprintf("%s/v1/webhooks/efi", s.baseURL)
	
	payload, _ := json.Marshal(map[string]interface{}{
		"event": "pix_received",
		"txid":  txHash,
		"value": 100000, // R$ 1.000,00 em sub-unidades (int64)
	})

	const workers = 40
	var wg sync.WaitGroup
	var mu sync.Mutex
	success, failure := 0, 0
	trigger := make(chan struct{})

	for i := 0; i < workers; i++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			<-trigger // Sincronização atômica

			req, _ := http.NewRequestWithContext(ctx, "POST", url, bytes.NewBuffer(payload))
			req.Header.Set("Content-Type", "application/json")
			// Assinatura mTLS/Header simulada - tentando burlar via re-envio puro
			req.Header.Set("X-Efi-Signature", "fake_calculated_signature_attempt") 

			resp, err := s.client.Do(req)
			if err != nil {
				mu.Lock()
				failure++
				mu.Unlock()
				return
			}
			defer resp.Body.Close()

			mu.Lock()
			if resp.StatusCode == http.StatusOK || resp.StatusCode == http.StatusAccepted {
				success++
			} else {
				failure++
			}
			mu.Unlock()
		}()
	}

	close(trigger)
	wg.Wait()
	return success, failure
}

// ---------------------------------------------------------------------
// 2. MODULO MACHINE LAYER (MCP): Ataque de Inundação de Contexto e Chaves Mutantes
// ---------------------------------------------------------------------
// Tenta exaurir os limites de tokens da OpenAI e estourar as tabelas de 
// subscrição do MCP, simulando um agente de IA malicioso que sofreu exploit.
func (s *HybridAdversarialSuite) AttackMCPContextExhaustion(ctx context.Context, apiKey string) int {
	url := fmt.Sprintf("%s/mcp/tools/call", s.baseURL)
	
	// Payload gigante simulando estouro de buffer de contexto para IA
	payload, _ := json.Marshal(map[string]interface{}{
		"tool": "list_webhook_subscriptions",
		"arguments": map[string]string{
			"bloat_padding": string(make([]byte, 50000)), // 50KB de lixo estruturado
		},
	})

	const spamRequests = 100
	var wg sync.WaitGroup
	var mu sync.Mutex
	rateLimitedCount := 0

	for i := 0; i < spamRequests; i++ {
		wg.Add(1)
		go func(id int) {
			defer wg.Done()
			req, _ := http.NewRequestWithContext(ctx, "POST", url, bytes.NewBuffer(payload))
			req.Header.Set("X-Agent-API-Key", apiKey)
			
			// Gera hashes parciais para tentar confundir o rate limiter de sliding-window
			req.Header.Set("X-Idempotency-Key", fmt.Sprintf("mcp_flood_%d", id))

			resp, err := s.client.Do(req)
			if err != nil {
				return
			}
			defer resp.Body.Close()

			if resp.StatusCode == http.StatusTooManyRequests {
				mu.Lock()
				rateLimitedCount++
				mu.Unlock()
			}
		}(i)
	}

	wg.Wait()
	return rateLimitedCount
}

package adversarial

import (
	"context"
	"testing"
	"time"
)

func TestIntegratedHybridAdversarialSuite(t *testing.T) {
	ctx, cancel := context.WithTimeout(context.Background(), 15*time.Second)
	defer cancel()

	// Aponta para o ambiente local levantado para testes automatizados
	suite := NewHybridAdversarialSuite("https://api.chainfx.local")
	secureAgentKey := "prod_m2m_agent_secure_key_123"

	// --- TESTE 1: Bloqueio de Duplo Payout Fiduciário ---
	t.Run("Human Rail: Garantir que Webhook duplicado não gera duplo gasto", func(t *testing.T) {
		txHash := "0xabc1239876543210ffffffffffffffffffffffffffffffffffffffffffffffff"
		successes, failures := suite.AttackHumanRailWebhook(ctx, txHash)
		
		t.Logf("[Human Rail Attack] Processados com Sucesso: %d | Rejeitados/Bloqueados: %d", successes, failures)
		
		if successes > 1 {
			t.Errorf("🚨 FALHA DE CUSTÓDIA: O Human Rail aceitou a mesma notificação Pix de liquidação %d vezes simultâneas!", successes)
		}
	})

	// --- TESTE 2: Resiliência do Rate Limiter do MCP ---
	t.Run("Machine Layer: Validar se o rate-limiter barrou o estouro de infra de IA", func(t *testing.T) {
		blockedRequests := suite.AttackMCPContextExhaustion(ctx, secureAgentKey)
		t.Logf("[MCP Attack] Requisições derrubadas pelo Rate Limiter: %d/100", blockedRequests)
		
		if blockedRequests < 40 { // Considerando os Tiers configurados no seu hardening
			t.Errorf("🚨 BRECHA DE SEGURANÇA: O rate limiter permitiu requisições em massa acima do esperado pelo Tier")
		}
	})
}
